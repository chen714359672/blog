关于JS精度问题的处理

# 前言：

  1. 因为公司项目涉及到了虚拟商品，同时价格的话保留6位。在购物车结算的时候就会有常见的js精度问题了。
  2. 因为相加相乘，还有价格只能往上，保留6位的同时，要往第7位0舍1入，因此的话toFixed()是明显不行的了。
  3. 后面也是找了js工具库跟函数相结合，才搞定了这个问题，所以要记录一下

# 解决思路：

多个精度工具库比较后，最终决定使用** bignumber.js (opens new window)**，但单单这个的话只解决了精度问题，并不能使小数第7位1入0舍，所以结合了下面的函数方法

class Calculate {}

/**
 * 精确加
 * @param arg1
 * @param arg2
 * @returns {number}
 */
Calculate.add = function(arg1, arg2) {
  var r1 = 0; var r2 = 0; var r3 = 0
  try {
    r1 = (arg1 + '').split('.')[1].length
  } catch (err) {
    r1 = 0
  }
  try {
    r2 = (arg2 + '').split('.')[1].length
  } catch (err) {
    r2 = 0
  }
  r3 = Math.pow(10, Math.max(r1, r2))
  return (Calculate.mul(arg1, r3) + Calculate.mul(arg2, r3)) / r3
}
/**
 * 精确减
 * @param arg1
 * @param arg2
 * @returns {number}
 */
Calculate.sub = function(arg1, arg2) {
  var r1 = 0; var r2 = 0; var r3 = 0
  try {
    r1 = (arg1 + '').split('.')[1].length
  } catch (err) {
    r1 = 0
  }
  try {
    r2 = (arg2 + '').split('.')[1].length
  } catch (err) {
    r2 = 0
  }
  r3 = Math.pow(10, Math.max(r1, r2))
  return (Calculate.mul(arg1, r3) - Calculate.mul(arg2, r3)) / r3
}
/**
 * 精确乘
 * @param arg1
 * @param arg2
 * @returns {number}
 */
Calculate.mul = function(arg1, arg2) {
  var r1 = arg1 + ''; var r2 = arg2 + ''; var r3 = 0; var r4 = 0
  try {
    r3 = r1.split('.')[1].length
  } catch (err) {
    r3 = 0
  }
  try {
    r4 = r2.split('.')[1].length
  } catch (err) {
    r4 = 0
  }
  return Number(r1.replace('.', '')) * Number(r2.replace('.', '')) / Math.pow(10, r4 + r3)
}
/**
 * 精确除
 * @param arg1
 * @param arg2
 * @returns {number}
 */
Calculate.div = function(arg1, arg2) {
  var r1 = arg1 + ''; var r2 = arg2 + ''; var r3 = 0; var r4 = 0
  try {
    r3 = r1.split('.')[1].length
  } catch (err) {
    r3 = 0
  }
  try {
    r4 = r2.split('.')[1].length
  } catch (err) {
    r4 = 0
  }
  return Calculate.mul((Number(r1.replace('.', '')) / Number(r2.replace('.', ''))), Math.pow(10, r4 - r3))
}
/**
 * 精确取余
 * @param arg1
 * @param arg2
 * @returns {number}
 */
Calculate.rem = function(arg1, arg2) {
  var r1 = 0; var r2 = 0; var r3 = 0
  try {
    r1 = (arg1 + '').split('.')[1].length
  } catch (err) {
    r1 = 0
  }
  try {
    r2 = (arg2 + '').split('.')[1].length
  } catch (err) {
    r2 = 0
  }
  r3 = Math.pow(10, Math.max(r1, r2))
  return Calculate.mul(arg1, r3) % Calculate.mul(arg2, r3) / r3
}
/**
 * 精确乘方
 * @param arg1
 * @param arg2
 * @returns {number}
 */
Calculate.pow = function(arg1, arg2) {
  var r1 = 0; var r2 = 0
  try {
    r1 = (arg1 + '').split('.')[1].length
  } catch (err) {
    r1 = 0
  }
  r2 = Math.pow(10, r1)
  return Math.pow(Calculate.mul(arg1, r2), arg2) / Math.pow(r2, arg2)
}
/**
 * 精确保留小数
 * @param arg1
 * @param dig 小数位数
 * @param flag 保留方式（-1：去尾；0：四舍五入；1：上位）
 * @returns {string}
 */
Calculate.fix = function(arg1, dig, flag) {
  dig = +dig > 0 ? +dig : 0
  flag = Math.abs(flag) === 1 ? flag : 0
  var r1 = Math.pow(10, dig); var r2 = 0
  if (flag === -1) r2 = Math.floor(Calculate.mul(arg1, r1)) / r1
  if (flag === 0) r2 = Math.round(Calculate.mul(arg1, r1)) / r1
  if (flag === 1) r2 = Math.ceil(Calculate.mul(arg1, r1)) / r1
  return r2.toFixed(dig)
}

export default Calculate
